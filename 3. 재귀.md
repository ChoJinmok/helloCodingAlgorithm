### 반복문
```javascript
function lookForKey(mainBox){
  let pile = [];
  for(const item of mainBox){
    pile.push(item);
  }
  
  while(pile.length !== 0){
    const box = pile.pop();
    for(const item of box){
      if(item === 'key'){
        console.log("열쇠를 찾았어요!");
        break;
      }
      if(item.constructor === Array){
        pile.push(item);
      }
    }
  }
}

console.log(lookForKey([[[],'key'],[]]))
```

<br />

### 재귀
```javascript
function lookForKey(box){
  for(const item of box){
    if(item === 'key'){
      console.log("열쇠를 찾았어요!");
      break;
    }
    if(item.constructor === Array){
      lookForKey(item)
    }
  }
}

console.log(lookForKey([[[],'key'],[]]))
```

### 카운트다운
```javascript
function countdown(i){
  console.log(i);
  // 기본 단계 (없으면 무한 반복된)
  if(i <= 1) return;
  // 재귀 단계
  countdown(i-1); 
}

countdown(3);
```

### 팩토리얼!
```javascript
function fact(x){
  if(x === 1) return 1;
  return x * fact(x-1);
}

console.log(fact(3));
```
